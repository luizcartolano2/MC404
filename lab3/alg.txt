tamanho dos vetores --> 3FF
endereco do vetor 1 --> 3FD
endereco do vetor 2 --> 3FE
resposta final --> AC

variavel "somatorio" --> valor: 0 endereco: 40

-------- O que fazer --------
#000
LOAD M(3FF) #CARREGO O CONTADOR NO REGISTRADOR AC

#001

#003

#005

LOAD MQ,M #CARREGO O VALOR DO VETOR 1 NO REGISTRADOR MQ

#006

#007

#008
